
// SHOW WIEW ACCOUNT SQL BLOCKCHAIN
SELECT DISTINCT
       A.USERID 
     , A.USERNAME 
	 , B.BALANCE 
	 , C.COIN_ID 
	 , C.QUANTITY 
	 , b.ACCOUNTNAME 
	 , UF_PRICE_TO_MONEY(C.COIN_ID, C.QUANTITY) AS  moneyCoin
FROM USERS A , ACCOUNTS B , ACCOUNTCOIN C 
WHERE A.USERID = B.USERID 
AND   B.ADDRESS = C.ADDRESSCOIN
ORDER BY A.USERNAME
